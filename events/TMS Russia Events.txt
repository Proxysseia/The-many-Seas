country_event = {
	id = 198720
	title = "Erasure of the Yenisey"
	desc = "We will burn their institutions to the ground and rise the flag of the Tsar in their place."
	fire_only_once = yes
	
	trigger = {
		NOT = {
			exists = YEN
		}
		tag = URA
	}
	
	option = {
		name = "For the Tsar!"
		effect = {
			prestige = 2
			YEN = { any_owned_state = { add_core =  URA } }
			YEN = { any_owned_state = { remove_core = YEN } }
		}
	}
}
